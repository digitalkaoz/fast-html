Basic usage

when called without any arguments

should return undefined
var returnVal = parse();
assert.equal(returnVal, undefined);

Simple examples

when called with empty string

should return an empty root-node
var returnVal = parse('', createElementNode, createTextNode);
assert.equal(returnVal.tagName, null);
assert.deepEqual(returnVal.attributes, {});
assert.deepEqual(returnVal.childNodes, []);

when called with just one element

should parse the string and return the element as childNode of root-node
assert.equal(returnVal.childNodes.length, 1, 'create just one childNode');
assert.equal(returnVal.childNodes[0].tagName, 'div');
should have empty attributes and childNodes
assert.deepEqual(returnVal.childNodes[0].attributes, {});
assert.equal(returnVal.childNodes[0].childNodes.length, 0);

camelCase utility

when called with an empty string

should return an empty string
var returnVal = camelCase('');
assert.equal(returnVal, '');

when called with a kebapcased string

with one part

should return the same string
var returnVal = camelCase('kebap-');
assert.equal(returnVal, 'kebap-');

with two parts

should return a camelCased string
var returnVal = camelCase('kebap-case-string');
assert.equal(returnVal, 'kebapCaseString');

with n parts

should return a camelCased string
var returnVal = camelCase('kebap-case-string-kebap-case-string-kebap-case-string');
assert.equal(returnVal, 'kebapCaseStringKebapCaseStringKebapCaseString');

when called with a non-kepabcased string

should return the same string
var returnVal = camelCase('kebapcase');
assert.equal(returnVal, 'kebapcase');

getAttributes utility

when called with an empty string

should return an empty object
var returnVal = getAttributes('');
assert.deepEqual(returnVal, {});

when called with a string containing an empty attribute

should return an object containing the attribute key with value `true`
var returnVal = getAttributes('foo');
assert.propertyVal(returnVal, 'foo', true);

when called with a string containing attributes

without quotes

should return an corresponding object
var returnVal = getAttributes('foo=bar');
assert.propertyVal(returnVal, 'foo', 'bar');

with single quotes

should return an corresponding object
var returnVal = getAttributes('foo=\'bar\'');
assert.propertyVal(returnVal, 'foo', 'bar');

with double quotes

should return an corresponding object
var returnVal = getAttributes('foo="bar"');
assert.propertyVal(returnVal, 'foo', 'bar');

with spaces between delimeters

should return an corresponding object
var returnVal = getAttributes(' foo = "bar" ');
assert.propertyVal(returnVal, 'foo', 'bar');

without spaces between delimeters

should return an corresponding object
var returnVal = getAttributes(' foo="bar" ');
assert.propertyVal(returnVal, 'foo', 'bar');
